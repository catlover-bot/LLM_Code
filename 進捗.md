#include <stdio.h>
#include <stdbool.h>

// 素数かどうかを判定する関数
bool is_prime(int num) {
    if (num <= 1) return false; // 1以下は素数ではない
    if (num == 2) return true;  // 2は素数
    if (num % 2 == 0) return false; // 偶数は素数ではない

    // 3からnumの平方根までチェック
    for (int i = 3; i * i <= num; i += 2) {
        if (num % i == 0) {
            return false;
        }
    }
    return true;
}

int main() {
    printf("2, "); // 2は素数なので最初に出力

    for (int i = 3; i <= 100; i += 2) {
        if (is_prime(i)) {
            printf("%d, ", i);
        }
    }

    printf("\n"); // 改行を入れる
    return 0;
}
